    • h : towerM K ⧸ towerT K → towerM K ⧸ towerT K → ZMod 2
    • hsym (v w : towerM K ⧸ towerT K) : self.h v w = self.h w v
    • hdiag (v : towerM K ⧸ towerT K) : self.h v v = 0
    • hzero (w : towerM K ⧸ towerT K) : self.h 0 w = 0
    • hcoc (u v w : towerM K ⧸ towerT K) : self.h u v + self.h (u + v) w = self.h v w + self.h u (v + w)
